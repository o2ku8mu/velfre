        function renderBigFiveChart(stats, colorClass) {

                const widthPercent = Math.abs(value);

                    ? `left: 50%; width: ${widthPercent}%; border-radius: 0 4px 4px 0;`

                            <div class="bigfive-center-line"></div>

                `;

                container.innerHTML += html;

            }

        }
